
\section{Módulos}
\begin{frame}
\frametitle{Definição de Módulo} 

\begin{definicao}
O \sub{módulo} (ou \sub{valor absoluto}) de um número real $x$,
denotado por $\modu x$, é definido por:
$$
\modu x =
\begin{cases}
x , & \text{se $x\geq 0$} \\
-x, & \text{se $x<0$}.
\end{cases}
$$
\end{definicao}


\end{frame}

\begin{frame}
\frametitle{Equações Modulares} 

Para resolver equações modulares, usaremos dois métodos:
\begin{itemize}
	\item Eliminação do módulo pela definição;
	\item Partição em intervalos.
\end{itemize}

\begin{exemplo}
Resolva as equações
\begin{enumerate}[(a)]
	\item  $\modu {2x-5} = 3$;
	\item $\modu{2x-3} = 1-3x$;
	\item $\modu{3-x} - \modu{x+1} = 4$.
\end{enumerate}

\end{exemplo}


\end{frame}
